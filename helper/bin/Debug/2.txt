BEGIN
  FOR I IN (SELECT A.ROWID,B.SSSSIDCARDNUM,B.SSSSOBJECTNAME,B.HOMEADDRESS,B.ACCOUNTNUM FROM COUNTYDATA.WFBT A,V_BASEOBJECT B WHERE A.OID=B.OBJECTID) LOOP
  UPDATE COUNTYDATA.WFBT SET SSSSIDCARDNUM=I.SSSSIDCARDNUM,SSSSOBJECTNAME=I.SSSSOBJECTNAME,DZ=I.HOMEADDRESS,ACCOUNTNUM=I.ACCOUNTNUM WHERE ROWID=I.ROWID;
END LOOP;
END;
